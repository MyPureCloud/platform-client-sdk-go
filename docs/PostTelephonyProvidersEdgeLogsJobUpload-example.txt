config := platformclientv2.GetDefaultConfiguration()
config.BasePath = ""    // expected format: mypurecloud.com
config.AccessToken = "your_access_token" // or use AuthorizeClientCredentials(...) or AuthorizeCodeGrant(...)

apiInstance := platformclientv2.NewTelephonyProvidersEdgeApiWithConfig(config)

var edgeId string // Edge ID
var jobId string // Job ID
var body platformclientv2.Edgelogsjobuploadrequest // Log upload request
// Request that the specified fileIds be uploaded from the Edge.
response, err := apiInstance.PostTelephonyProvidersEdgeLogsJobUpload(edgeId, jobId, body)
fmt.Printf("Response:\n  Success: %v\n  Status code: %v\n  Correlation ID: %v\n", response.IsSuccess, response.StatusCode, response.CorrelationID)
if err != nil {
	fmt.Printf("Error calling PostTelephonyProvidersEdgeLogsJobUpload: %v\n", err)
}
