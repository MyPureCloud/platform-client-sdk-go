apiInstance := platformclientv2.NewWorkforceManagementApiWithConfig(config)

var weekDateId time.Time // The start date of the week schedule in yyyy-MM-dd format. Dates are represented as an ISO-8601 string. For example: yyyy-MM-dd
var tradeId string // The ID of the shift trade to update
