config := platformclientv2.GetDefaultConfiguration()

apiInstance := platformclientv2.NewIntegrationsApiWithConfig(config)

var nuanceIntegrationId string // The integration ID for this group of bots
var botId string // The Nuance bot ID to get
var expand []string // expand
var targetChannel string // targetChannel
// Get a Nuance bot in the specified Integration

}
