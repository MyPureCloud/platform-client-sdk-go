config := platformclientv2.GetDefaultConfiguration()
config.BasePath = ""    // expected format: mypurecloud.com
config.AccessToken = "your_access_token" // or use AuthorizeClientCredentials(...) or AuthorizeCodeGrant(...)

apiInstance := platformclientv2.NewScriptsApiWithConfig(config)

var pageSize int // Page size
var pageNumber int // Page number
var expand string // Expand
var name string // Name filter
var feature string // Feature filter
var flowId string // Secure flow id filter
var scriptDataVersion string // Advanced usage - controls the data version of the script
var divisionIds string // Filters scripts to requested divisionIds
// Get the published scripts.
data, response, err := apiInstance.GetScriptsPublished(pageSize, pageNumber, expand, name, feature, flowId, scriptDataVersion, divisionIds)
fmt.Printf("Response:\n  Success: %v\n  Status code: %v\n  Correlation ID: %v\n  Body: %v\n", response.IsSuccess, response.StatusCode, response.CorrelationID, data.String())
if err != nil {
	fmt.Printf("Error calling GetScriptsPublished: %v\n", err)
}
