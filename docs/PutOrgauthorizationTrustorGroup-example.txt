config := platformclientv2.GetDefaultConfiguration()
config.BasePath = ""    // expected format: mypurecloud.com
config.AccessToken = "your_access_token" // or use AuthorizeClientCredentials(...) or AuthorizeCodeGrant(...)

apiInstance := platformclientv2.NewOrganizationAuthorizationApiWithConfig(config)

var trustorOrgId string // Trustor Organization Id
var trustorGroupId string // Trustor Group Id
// Add a Trustee Group to the trust.
data, response, err := apiInstance.PutOrgauthorizationTrustorGroup(trustorOrgId, trustorGroupId)
fmt.Printf("Response:\n  Success: %v\n  Status code: %v\n  Correlation ID: %v\n  Body: %v\n", response.IsSuccess, response.StatusCode, response.CorrelationID, data.String())
if err != nil {
	fmt.Printf("Error calling PutOrgauthorizationTrustorGroup: %v\n", err)
}
